    def exp(cls, iteration: int, t0: float, count: int, limit: int) -> float:
        """Exponential temperature function."""
        k = count / limit
        temperature = t0 * (0.95 ** k)
        _LOGGER.debug(
            "New temperature for (iteration=%d, t0=%d, count=%d, limit=%d, k=%f) = %g",
            iteration,
            t0,
            count,
            limit,
            k,
            temperature,
        )
        return temperature
